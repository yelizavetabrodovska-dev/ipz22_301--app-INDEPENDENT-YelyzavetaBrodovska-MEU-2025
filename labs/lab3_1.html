function filterArray(numbers, value) {
  const filteredNumbers = [];

  for (let i = 0; i < numbers.length; i++) {
    if (numbers[i] > value) {
      filteredNumbers.push(numbers[i]);
    }
  }

  return filteredNumbers;
}

function makeArray(firstArray, secondArray, maxLength) {
  const newArray = firstArray.concat(secondArray);

  if (newArray.length > maxLength) {
    return newArray.slice(0, maxLength);
  }

  return newArray;
}

// Функція для створення масиву заданої довжини з випадковими числами
function createArray(length) {
  const arr = [];
  for (let i = 0; i < length; i++) {
    arr.push(Math.floor(Math.random() * 100)); // випадкові числа 0–99
  }
  return arr;
}

// Функція для знаходження максимального серед парних елементів
function findMaxEvenElement(arr) {
  let maxEven = null;
  for (let i = 0; i < arr.length; i++) {
    if (arr[i] % 2 === 0) {
      if (maxEven === null || arr[i] > maxEven) {
        maxEven = arr[i];
      }
    }
  }
  return maxEven;
}

// Функція для знаходження мінімального серед елементів із парними індексами
function findMinEvenIndexElement(arr) {
  let minEvenIndex = arr[0];
  for (let i = 0; i < arr.length; i += 2) {
    if (arr[i] < minEvenIndex) {
      minEvenIndex = arr[i];
    }
  }
  return minEvenIndex;
}

// Функція для обміну місцями двох елементів у масиві
function swapElements(arr, val1, val2) {
  const index1 = arr.indexOf(val1);
  const index2 = arr.indexOf(val2);
  if (index1 !== -1 && index2 !== -1) {
    const temp = arr[index1];
    arr[index1] = arr[index2];
    arr[index2] = temp;
  }
  return arr;
}

// Функція для сортування масиву методом вставки
function insertionSort(arr) {
  for (let i = 1; i < arr.length; i++) {
    let current = arr[i];
    let j = i - 1;
    while (j >= 0 && arr[j] > current) {
      arr[j + 1] = arr[j];
      j--;
    }
    arr[j + 1] = current;
  }
  return arr;
}

// --- Перевірка чисел ---
const isInteger = (s) => /^[-+]?\d+$/.test(s.trim());
const isFloat = (s) => /^[-+]?((\d+\.\d+)|(\d+\.)|(\.\d+)|(\d+))$/.test(s.trim());

// --- Перевірка дати у форматі DD.MM.YYYY ---
function parseDateDDMMYYYY(s) {
  const m = s.trim().match(/^(\d{2})\.(\d{2})\.(\d{4})$/);
  if (!m) return null;

  const day = Number(m[1]);
  const month = Number(m[2]);
  const year = Number(m[3]);

  if (month < 1 || month > 12) return null;
  const mdays = [31, (isLeap(year) ? 29 : 28), 31, 30, 31, 30, 31, 31, 30, 31, 30, 31];
  if (day < 1 || day > mdays[month - 1]) return null;

  return { day, month, year };
}

function isLeap(y) {
  return (y % 4 === 0 && y % 100 !== 0) || (y % 400 === 0);
}

// --- Відображення / очищення помилок ---
function showError(name, msg) {
  const el = document.querySelector(`.error[data-for="${name}"]`);
  const input = document.getElementById(name);
  if (msg) {
    el.textContent = msg;
    input.parentElement.classList.add('has-error');
  } else {
    el.textContent = '';
    input.parentElement.classList.remove('has-error');
  }
}

// --- Валідатори полів ---
const validators = {
  username: (v) => {
    if (!v) return "Поле обов'язкове.";
    if (v.length < 3 || v.length > 30) return "Ім'я користувача має бути від 3 до 30 символів.";
    return '';
  },
  email: (v) => {
    if (!v) return "Поле обов'язкове.";
    const re = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
    if (!re.test(v)) return 'Невірний формат email.';
    return '';
  },
  age: (v) => {
    if (!v) return '';
    if (!isInteger(v)) return 'Потрібне ціле число.';
    return '';
  },
  height: (v) => {
    if (!v) return '';
    if (!isFloat(v)) return 'Потрібне дійсне число (десяткова крапка ".").';
    return '';
  },
  birthdate: (v) => {
    if (!v) return "Поле обов'язкове.";
    if (!parseDateDDMMYYYY(v)) return 'Невірна дата. Використайте формат DD.MM.YYYY.';
    return '';
  },
  password: (v) => {
    if (!v) return "Поле обов'язкове.";
    if (v.length < 6) return 'Мінімум 6 символів.';
    return '';
  },
  confirm_password: (v) => {
    const pw = document.getElementById('password').value || '';
    if (!v) return "Поле обов'язкове.";
    if (v !== pw) return 'Паролі не співпадають.';
    return '';
  }
};

// --- Перевірка одного поля ---
function validateField(name) {
  const input = document.getElementById(name);
  const fn = validators[name];
  if (!fn) return true;
  const err = fn(input.value);
  showError(name, err);
  return !err;
}

// --- Реакція на події введення ---
document.querySelectorAll('#regForm input').forEach(el => {
  el.addEventListener('input', () => validateField(el.id));
  el.addEventListener('blur', () => validateField(el.id));
});

// --- Обробка відправлення форми ---
document.getElementById('regForm').addEventListener('submit', function (e) {
  e.preventDefault();
  const names = ['username', 'email', 'age', 'height', 'birthdate', 'password', 'confirm_password'];
  let firstInvalid = null;
  let ok = true;

  names.forEach(name => {
    const valid = validateField(name);
    if (!valid) {
      ok = false;
      if (!firstInvalid) firstInvalid = document.getElementById(name);

  const result = document.getElementById('result');
  if (!ok) {
    result.innerHTML = '
Будь ласка, виправте помилки у формі.
';
    firstInvalid.focus();

  }

  const data = {
    username: this.username.value.trim(),
    email: this.email.value.trim(),
    age: this.age.value.trim() || null,
    height: this.height.value.trim() || null,
    birthdate: this.birthdate.value.trim()
  };

  result.innerHTML = `
    

      Реєстрація успішна!

      Надіслані дані (демо):
      
${JSON.stringify(data, null, 2)}

    
`;
  this.reset();
});

Висновки

У ході виконання роботи було створено HTML-форму реєстрації користувача з перевіркою правильності введення даних.
Було реалізовано такі етапи:

  • Створено структуру форми з обов’язковими полями: ім’я користувача, вік, дата народження, пароль та підтвердження пароля.
  • Реалізовано клієнтську валідацію засобами JavaScript:
    • перевірка введення цілого числа (для віку);
    • перевірка дійсного числа (для зарплати);
    • перевірка дати у форматі DD.MM.YYYY;
    • перевірка співпадіння паролів;
    • контроль заповнення обов’язкових полів, позначених символом (*).
  • Додано виведення повідомлень про помилки біля відповідних полів.
  • У разі успішного заповнення форми виводиться коротке зведення введених даних.

Результат:

Було розроблено функціональну форму з гнучкою перевіркою введення, що забезпечує коректність даних перед їх надсиланням на сервер. Отримані знання можна застосовувати для створення реальних веб-форм реєстрації, авторизації або зворотного зв’язку.
