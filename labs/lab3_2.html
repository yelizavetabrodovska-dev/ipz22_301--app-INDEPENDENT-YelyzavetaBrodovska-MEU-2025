const medicines = {
  Агалгін: new Date("2022-05-01"),

  Альфахолін: new Date("2024-12-21"),
  Аспірин: new Date("2022-08-15"),
  Аспаркам: new Date("2024-04-18"),
};

// Отримати лише назви препаратів, що ще не прострочені
const currentDate = new Date();
const validMedicines = Object.entries(medicines)
  .filter(([name, date]) => date > currentDate)
  .sort((a, b) => a[1] - b[1])
  .map(([name]) => name);

console.log(validMedicines);

const fruits = [
  { name: "apple", price: 200 },
  { name: "orange", price: 300 },
  { name: "grapes", price: 750 },
];

// Функція, що повертає новий масив з id і знижкою 20%
function updateFruits(fruits) {
  return fruits.map((fruit, index) => ({
    id: index + 1,
    name: fruit.name,
    price: fruit.price * 0.8, // 20% знижка
  }));
}

const discountedFruits = updateFruits(fruits);
console.log(discountedFruits);

const tweets = [
  { id: "000", likes: 5, tags: ["js", "nodejs"] },
  { id: "001", likes: 2, tags: ["html", "css"] },
  { id: "002", likes: 17, tags: ["html", "js", "nodejs"] },
  { id: "003", likes: 8, tags: ["css", "react"] },
  { id: "004", likes: 0, tags: ["js", "nodejs", "react"] },
];

// Повертаємо об'єкт із кількістю кожного тегу
const tagCount = tweets
  .flatMap(tweet => tweet.tags)
  .reduce((acc, tag) => {
    acc[tag] = (acc[tag] || 0) + 1;
    return acc;
  }, {});

console.log(tagCount);

function checkBrackets(str) {
  const stack = [];
  const brackets = {
    "(": ")",
    "[": "]",
    "{": "}"
  };

  for (const char of str) {
    if (brackets[char]) {
      stack.push(char);
    } else if (Object.values(brackets).includes(char)) {
      const last = stack.pop();
      if (brackets[last] !== char) return false;
    }
  }
  return stack.length === 0;
}

console.log(checkBrackets("function test() { return [1, 2, 3]; }"));
console.log(checkBrackets("function bad() { return [1, 2, 3; }"));
